@Component({
  selector: 'app-series',
  templateUrl: './series.page.html',
  styleUrls: ['./series.page.scss'],
})
export class SeriesPage implements OnInit {
  isAdmin = false;
  ports: Port[] = [];
  port!: Port;
  page_ = 2;
  portsSubscription!: Subscription;
  _series: BehaviorSubject<Serie[]> = new BehaviorSubject<Serie[]>([]);
  series$: Observable<Serie[]> = this._series.asObservable();
  _filteredSeries: BehaviorSubject<Serie[]> = new BehaviorSubject<Serie[]>([]);
  filteredSeries$: Observable<Serie[]> = this._filteredSeries.asObservable();
  searchTerm: string = '';

  public alertYesNoButtons = [
    {
      text: 'No',
      role: 'no',
    },
    {
      text: 'Yes',
      role: 'yes',
    },
  ];
  isWeb: boolean = false;
  private loadedIds: Set<string> = new Set();

  constructor(
    private animationCtrl: AnimationController,
    private seasonsSvc: SeasonsService,
    private serieSvc: SeriesService,
    private modalCtrl: ModalController,
    private translate: TranslateService,
    private alertCtrl: AlertController,
    private platform: Platform,
    private authSvc: FirebaseAuthenticationService,
    @Inject(SERIES_COLLECTION_SUBSCRIPTION_TOKEN)
    private seriesSubscription: ICollectionSubscription<Serie>
  ) {
    this.isWeb = this.platform.is('desktop');
  }

  ngOnInit(): void {
    this.loadSeries();

    this.authSvc
      .getCurrentUser()
      .then((user) => {
        this.isAdmin = user.admin;
      })
      .catch(() => {
        this.isAdmin = false;
      });

    this._series.subscribe((series) => {
      this._filteredSeries.next(series);
    });

    this.seriesSubscription
      .subscribe('series')
      .subscribe((change: CollectionChange<Serie>) => {
        const currentSeries = [...this._series.value];

        if (!this.loadedIds.has(change.id) && change.type !== 'added') {
          return;
        }

        switch (change.type) {
          case 'added':
            this.loadedIds.add(change.id);

            const existingIndex = currentSeries.findIndex(
              (p) => p.id === change.id
            );
            if (existingIndex === -1) {
              currentSeries.push(change.data!);
              currentSeries.sort((a, b) =>
                (a.name || '').localeCompare(b.name || '')
              );
            }
            break;
          case 'modified':
            const index = currentSeries.findIndex((p) => p.id === change.id);
            if (index >= 0) {
              currentSeries[index] = change.data!;
              currentSeries.sort((a, b) =>
                (a.name || '').localeCompare(b.name || '')
              );
            }
            break;
          case 'removed':
            const removeIndex = currentSeries.findIndex(
              (p) => p.id === change.id
            );
            if (removeIndex >= 0) {
              currentSeries.splice(removeIndex, 1);
              this.loadedIds.delete(change.id);
            }
            break;
        }

        this._series.next(currentSeries);
      });
  }

  @ViewChildren('avatar') avatars!: QueryList<ElementRef>;
  @ViewChild('animatedAvatar') animatedAvatar!: ElementRef;
  @ViewChild('animatedAvatarContainer') animatedAvatarContainer!: ElementRef;

  selectedSerie: any = null;
  isAnimating = false;
  page: number = 1;
  pageSize: number = 25;
  pages: number = 0;

  onSearch(event: any) {
    const searchTerm = event.target.value.toLowerCase();
    this.searchTerm = searchTerm;

    if (!searchTerm) {
      this._filteredSeries.next(this._series.value);
      return;
    }

    const filtered = this._series.value.filter((serie) =>
      serie.name.toLowerCase().includes(searchTerm)
    );

    this._filteredSeries.next(filtered);
  }

  loadSeries() {
    this.page = 1;
    this.serieSvc.getAll(this.page, this.pageSize).subscribe({
      next: (response: Paginated<Serie>) => {
        response.data.forEach((serie) => this.loadedIds.add(serie.id));

        const sortedSeries = response.data.sort((a, b) =>
          (a.name || '').localeCompare(b.name || '')
        );

        this._series.next([...sortedSeries]);
        this._filteredSeries.next([...sortedSeries]);
        this.page++;
        this.pages = response.pages;
      },
    });
  }

  loadMoreSeries(notify: HTMLIonInfiniteScrollElement | null = null) {
    if (this.page <= this.pages) {
      this.serieSvc.getAll(this.page, this.pageSize).subscribe({
        next: (response: Paginated<Serie>) => {
          response.data.forEach((serie) => this.loadedIds.add(serie.id));

          const updatedSeries = [...this._series.value, ...response.data].sort(
            (a, b) => (a.name || '').localeCompare(b.name || '')
          );
          this._series.next(updatedSeries);

          if (this.searchTerm) {
            const filtered = updatedSeries.filter((serie) =>
              serie.name.toLowerCase().includes(this.searchTerm)
            );
            this._filteredSeries.next(filtered);
          } else {
            this._filteredSeries.next(updatedSeries);
          }
          this.page++;
          notify?.complete();
        },
      });
    } else {
      notify?.complete();
    }
  }

  async openSerieDetail(serie: any, index: number) {
    await this.presentModalSerie('edit', serie);
    this.selectedSerie = serie;
  }

  onIonInfinite(ev: InfiniteScrollCustomEvent) {
    this.loadMoreSeries(ev.target);
  }

  private async presentModalSerie(
    mode: 'new' | 'edit',
    serie: Serie | undefined = undefined
  ) {
    let _series: Serie[] = await lastValueFrom(this.serieSvc.getAll());
    const modal = await this.modalCtrl.create({
      component: SerieModalComponent,
      componentProps:
        mode == 'edit'
          ? {
              serie: serie,
              series: _series,
            }
          : {
              series: _series,
            },
    });
    modal.onDidDismiss().then((response: any) => {
      switch (response.role) {
        case 'new':
          this.serieSvc.add(response.data).subscribe({
            next: (res) => {
              this.loadSeries();
            },
            error: (err) => {},
          });
          break;
        case 'edit':
          this.serieSvc.update(serie!.id, response.data).subscribe({
            next: (res) => {
              this.loadSeries();
            },
            error: (err) => {},
          });
          break;
        default:
          break;
      }
    });
    await modal.present();
  }

  async onAddSerie() {
    await this.presentModalSerie('new');
  }

  async onDeleteSerie(serie: Serie) {
    const alert = await this.alertCtrl.create({
      header: await this.translate
        .get('SERIES.MESSAGES.DELETE_CONFIRM')
        .toPromise(),
      cssClass: 'custom-delete-alert',
      buttons: [
        {
          text: 'Cancel',
          role: 'cancel',
        },
        {
          text: 'OK',
          role: 'yes',
          handler: () => {
            this.serieSvc.delete(serie.id).subscribe({
              next: (response) => {
                this.loadSeries();
              },
              error: (err) => {},
            });
          },
        },
      ],
    });

    await alert.present();
  }
}
