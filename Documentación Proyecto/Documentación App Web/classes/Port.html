  loadSeries(){
    this.page = 1;
    this.serieSvc.getSeriesWithSeasons(this.page, this.pageSize).subscribe({
      next:(response:Paginated<Serie>)=>{

        const seriesWithSortedSeasons = response.data.map(serie => ({
          ...serie,
          seasons: serie.seasons?.sort((a, b) => 
            (a.name || '').localeCompare(b.name || '')
          )
        }));

        this._series.next([...seriesWithSortedSeasons]);
        this.page++;
        this.pages = response.pages;
      }
    });
  }

  loadMoreSeries(notify:HTMLIonInfiniteScrollElement | null = null) {
    if(this.page <= this.pages){
      this.serieSvc.getSeriesWithSeasons(this.page, this.pageSize).subscribe({
        next:(response:Paginated<Serie>)=>{

          const seriesWithSortedSeasons = response.data.map(serie => ({
            ...serie,
            seasons: serie.seasons?.sort((a, b) => 
              (a.name || '').localeCompare(b.name || '')
            )
          }));

          this._series.next([...this._series.value, ...seriesWithSortedSeasons]);
          this.page++;
          notify?.complete();
        }
      });
    }
    else{
      notify?.complete();
    }
  }

  loadSeasons(){
    this.page=1;
    this.seasonsSvc.getAll(this.page, this.pageSize).subscribe({
      next:(response:Paginated<Season>)=>{
        this._seasons.next([...response.data]);
        this.page++;
        this.pages = response.pages;
      }
    });
  }


  loadMoreSeasons(notify:HTMLIonInfiniteScrollElement | null = null) {
    if(this.page<=this.pages){
      this.seasonsSvc.getAll(this.page, this.pageSize).subscribe({
        next:(response:Paginated<Season>)=>{
          this._seasons.next([...this._seasons.value, ...response.data]);
          this.page++;
          notify?.complete();
        }
      });
    }
    else{
      notify?.complete();
    }
  }
