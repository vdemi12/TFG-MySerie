@Component({
  selector: 'app-seasons',
  templateUrl: './seasons.page.html',
  styleUrls: ['./seasons.page.scss'],
})
export class SeasonsPage implements OnInit {
  isAdmin = false;
  ports: Port[] = [];
  port!: Port;
  page_ = 2;
  portsSubscription!: Subscription;
  _seasons: BehaviorSubject<Season[]> = new BehaviorSubject<Season[]>([]);
  seasons$: Observable<Season[]> = this._seasons.asObservable();
  _filteredSeasons: BehaviorSubject<Season[]> = new BehaviorSubject<Season[]>(
    []
  );
  filteredSeasons$: Observable<Season[]> = this._filteredSeasons.asObservable();
  searchTerm: string = '';

  public alertYesNoButtons = [
    {
      text: 'No',
      role: 'no',
    },
    {
      text: 'Yes',
      role: 'yes',
    },
  ];
  isWeb: boolean = false;
  private loadedIds: Set<string> = new Set();

  constructor(
    private animationCtrl: AnimationController,
    private seasonsSvc: SeasonsService,
    private serieSvc: SeriesService,
    private modalCtrl: ModalController,
    private translate: TranslateService,
    private alertCtrl: AlertController,
    private platform: Platform,
    private authSvc: FirebaseAuthenticationService,
    @Inject(SEASONS_COLLECTION_SUBSCRIPTION_TOKEN)
    private seasonsSubscription: ICollectionSubscription<Season>
  ) {
    this.isWeb = this.platform.is('desktop');
  }

  ngOnInit(): void {
    this.loadSeasons();

    this.authSvc
      .getCurrentUser()
      .then((user) => {
        this.isAdmin = user.admin;
      })
      .catch(() => {
        this.isAdmin = false;
      });

    this._seasons.subscribe((seasons) => {
      this._filteredSeasons.next(seasons);
    });

    this.seasonsSubscription
      .subscribe('seasons')
      .subscribe((change: CollectionChange<Season>) => {
        const currentSeasons = [...this._seasons.value];

        // Solo procesar cambios de documentos que ya tenemos cargados
        if (!this.loadedIds.has(change.id) && change.type !== 'added') {
          return;
        }

        switch (change.type) {
          case 'added':
            // Registrar el nuevo ID en nuestro conjunto de IDs cargados
            this.loadedIds.add(change.id);

            // Verificar si ya existe en nuestro array local
            const existingIndex = currentSeasons.findIndex(
              (p) => p.id === change.id
            );
            if (existingIndex === -1) {
              currentSeasons.push(change.data!);
              currentSeasons.sort((a, b) =>
                (a.name || '').localeCompare(b.name || '')
              );
            }
            break;
          case 'modified':
            const index = currentSeasons.findIndex((p) => p.id === change.id);
            if (index >= 0) {
              currentSeasons[index] = change.data!;
              currentSeasons.sort((a, b) =>
                (a.name || '').localeCompare(b.name || '')
              );
            }
            break;
          case 'removed':
            const removeIndex = currentSeasons.findIndex(
              (p) => p.id === change.id
            );
            if (removeIndex >= 0) {
              currentSeasons.splice(removeIndex, 1);
              this.loadedIds.delete(change.id);
            }
            break;
        }

        this._seasons.next(currentSeasons);
      });
  }

  @ViewChildren('avatar') avatars!: QueryList<ElementRef>;
  @ViewChild('animatedAvatar') animatedAvatar!: ElementRef;
  @ViewChild('animatedAvatarContainer') animatedAvatarContainer!: ElementRef;

  selectedSeason: any = null;
  isAnimating = false;
  page: number = 1;
  pageSize: number = 25;
  pages: number = 0;

  onSearch(event: any) {
    const searchTerm = event.target.value.toLowerCase();
    this.searchTerm = searchTerm;

    if (!searchTerm) {
      this._filteredSeasons.next(this._seasons.value);
      return;
    }

    const filtered = this._seasons.value.filter((season) =>
      season.name.toLowerCase().includes(searchTerm)
    );

    this._filteredSeasons.next(filtered);
  }

  loadSeasons() {
    this.page = 1;
    this.seasonsSvc.getAll(this.page, this.pageSize).subscribe({
      next: (response: Paginated<Season>) => {
        // Actualizar el registro de IDs cargados
        response.data.forEach((season) => this.loadedIds.add(season.id));

        const sortedSeasons = response.data.sort((a, b) =>
          (a.name || '').localeCompare(b.name || '')
        );

        this._seasons.next([...sortedSeasons]);
        this._filteredSeasons.next([...sortedSeasons]);
        this.page++;
        this.pages = response.pages;
      },
    });
  }

  loadMoreSeasons(notify: HTMLIonInfiniteScrollElement | null = null) {
    if (this.page <= this.pages) {
      this.seasonsSvc.getAll(this.page, this.pageSize).subscribe({
        next: (response: Paginated<Season>) => {
          // Actualizar el registro de IDs cargados
          response.data.forEach((season) => this.loadedIds.add(season.id));

          const updatedSeasons = [
            ...this._seasons.value,
            ...response.data,
          ].sort((a, b) => (a.name || '').localeCompare(b.name || ''));
          this._seasons.next(updatedSeasons);

          if (this.searchTerm) {
            const filtered = updatedSeasons.filter((season) =>
              season.name.toLowerCase().includes(this.searchTerm)
            );
            this._filteredSeasons.next(filtered);
          } else {
            this._filteredSeasons.next(updatedSeasons);
          }
          this.page++;
          notify?.complete();
        },
      });
    } else {
      notify?.complete();
    }
  }

  async openSeasonDetail(season: any, index: number) {
    await this.presentModalSeason('edit', season);
    this.selectedSeason = season;
  }

  onIonInfinite(ev: InfiniteScrollCustomEvent) {
    this.loadMoreSeasons(ev.target);
  }

  private async presentModalSeason(
    mode: 'new' | 'edit',
    season: Season | undefined = undefined
  ) {
    let _series: Serie[] = await lastValueFrom(this.serieSvc.getAll());
    const modal = await this.modalCtrl.create({
      component: SeasonModalComponent,
      componentProps:
        mode == 'edit'
          ? {
              season: season,
              series: _series,
            }
          : {
              series: _series,
            },
    });
    modal.onDidDismiss().then((response: any) => {
      switch (response.role) {
        case 'new':
          this.seasonsSvc.add(response.data).subscribe({
            next: (res) => {
              this.loadSeasons();
            },
            error: (err) => {},
          });
          break;
        case 'edit':
          this.seasonsSvc.update(season!.id, response.data).subscribe({
            next: (res) => {
              this.loadSeasons();
            },
            error: (err) => {},
          });
          break;
        default:
          break;
      }
    });
    await modal.present();
  }

  async onAddSeason() {
    await this.presentModalSeason('new');
  }

  async onDeleteSeason(season: Season) {
    const alert = await this.alertCtrl.create({
      header: await this.translate
        .get('SEASONS.MESSAGES.DELETE_CONFIRM')
        .toPromise(),
      cssClass: 'custom-delete-alert',
      buttons: [
        {
          text: 'Cancel',
          role: 'cancel',
        },
        {
          text: 'OK',
          role: 'yes',
          handler: () => {
            this.seasonsSvc.delete(season.id).subscribe({
              next: (response) => {
                this.loadSeasons();
              },
              error: (err) => {},
            });
          },
        },
      ],
    });

    await alert.present();
  }
}
