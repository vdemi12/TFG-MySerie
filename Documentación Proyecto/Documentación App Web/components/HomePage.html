import { TranslateService } from '@ngx-translate/core';
import { SeasonViewComponent } from 'src/app/shared/components/season-view/season-view.component';

@Component({
  selector: 'app-home',
  templateUrl: './home.page.html',
  styleUrls: ['./home.page.scss'],
})
export class HomePage implements OnInit {
  ports: Port[] = [];
  port!: Port;
  page_ = 2;
  portsSubscription!: Subscription;
  _series:BehaviorSubject<Serie[]> = new BehaviorSubject<Serie[]>([]);
  series$:Observable<Serie[]> = this._series.asObservable();
  _seasons:BehaviorSubject<Season[]> = new BehaviorSubject<Season[]>([]);
  seasons$:Observable<Season[]> = this._seasons.asObservable();

  isWeb: boolean = false;

  constructor(
    private seasonsSvc:SeasonsService,
    private serieSvc:SeriesService,
    private platform: Platform,
    private modalCtrl:ModalController,
    private translate: TranslateService,
  ) {
    this.isWeb = this.platform.is('desktop');
  }

  ngOnInit(): void {
    this.loadSeries();
    this.loadSeasons();
  }

  @ViewChildren('avatar') avatars!: QueryList<ElementRef>;
  @ViewChild('animatedAvatar') animatedAvatar!: ElementRef;
  @ViewChild('animatedAvatarContainer') animatedAvatarContainer!: ElementRef;

  selectedSeason: any = null;
  selectedSerie: any = null;
  isAnimating = false;
  page:number = 1;
  pageSize:number = 25;
  pages:number = 0;


  loadSeries(){
    this.page = 1;
    this.serieSvc.getSeriesWithSeasons(this.page, this.pageSize).subscribe({
      next:(response:Paginated<Serie>)=>{

        const seriesWithSortedSeasons = response.data.map(serie => ({
          ...serie,
          seasons: serie.seasons?.sort((a, b) => 
            (a.name || '').localeCompare(b.name || '')
          )
        }));

        this._series.next([...seriesWithSortedSeasons]);
        this.page++;
        this.pages = response.pages;
      }
    });
  }

  loadMoreSeries(notify:HTMLIonInfiniteScrollElement | null = null) {
    if(this.page <= this.pages){
      this.serieSvc.getSeriesWithSeasons(this.page, this.pageSize).subscribe({
        next:(response:Paginated<Serie>)=>{

          const seriesWithSortedSeasons = response.data.map(serie => ({
            ...serie,
            seasons: serie.seasons?.sort((a, b) => 
              (a.name || '').localeCompare(b.name || '')
            )
          }));

          this._series.next([...this._series.value, ...seriesWithSortedSeasons]);
          this.page++;
          notify?.complete();
        }
      });
    }
    else{
      notify?.complete();
    }
  }

  loadSeasons(){
    this.page=1;
    this.seasonsSvc.getAll(this.page, this.pageSize).subscribe({
      next:(response:Paginated<Season>)=>{
        this._seasons.next([...response.data]);
        this.page++;
        this.pages = response.pages;
      }
    });
  }


  loadMoreSeasons(notify:HTMLIonInfiniteScrollElement | null = null) {
    if(this.page<=this.pages){
      this.seasonsSvc.getAll(this.page, this.pageSize).subscribe({
        next:(response:Paginated<Season>)=>{
          this._seasons.next([...this._seasons.value, ...response.data]);
          this.page++;
          notify?.complete();
        }
      });
    }
    else{
      notify?.complete();
    }
  }
  async openSeasonDetail(season: any) {
    await this.presentModalSeason(season);
    this.selectedSeason = season;
  }
  private async presentModalSeason(season:Season|undefined=undefined){
    const modal = await this.modalCtrl.create({
      component: SeasonViewComponent,
      componentProps: {
        season: season
      }
    });
  
    await modal.present();
  }

  async openSerieDetail(serie: any) {
    await this.presentModalSerie(serie);
    this.selectedSerie = serie;
  }
  private async presentModalSerie(serie:Serie|undefined=undefined){
    const modal = await this.modalCtrl.create({
      component: SerieViewComponent,
      componentProps: {
        serie: serie
      }
    });
  
    await modal.present();
  }

  async readSerieTitle(serieTitle: string) {
    await ScreenReader.speak({ value: serieTitle });
  }
  
  onIonInfinite(ev:InfiniteScrollCustomEvent) {
    this.loadMoreSeries(ev.target);
    
  }
}
