import { Component, Input, OnInit } from '@angular/core';
import { FormBuilder, FormGroup, Validators } from '@angular/forms';
import { ModalController, Platform, LoadingController, ToastController } from '@ionic/angular';
import { CohereService } from 'src/app/core/services/impl/ai.service';
import { Serie } from 'src/app/core/models/serie.model';
import { LanguageService } from 'src/app/core/services/language.service';
import { TranslateService } from '@ngx-translate/core';
import { lastValueFrom } from 'rxjs';

@Component({
  selector: 'app-serie-modal',
  templateUrl: './serie-modal.component.html',
  styleUrls: ['./serie-modal.component.scss'],
})
export class SerieModalComponent  implements OnInit {
  formGroup:FormGroup;
  mode:'new'|'edit' = 'new';
  isMobile: boolean = false;
  isGeneratingSynopsis: boolean = false;

  @Input() set serie(_serie:Serie){
    if(_serie && _serie.id)
      this.mode = 'edit';
    
    this.formGroup.controls['name'].setValue(_serie.name);
    this.formGroup.controls['synopsis'].setValue(_serie.synopsis);
    this.formGroup.controls['releaseYear'].setValue(_serie.releaseYear);
    this.formGroup.controls['rating'].setValue(_serie.rating);
    if (_serie.picture) {
      this.formGroup.controls['picture'].setValue(_serie.picture.url);
    }
  }

  constructor(
    private fb: FormBuilder,
    private modalCtrl: ModalController,
    private platform: Platform,
    private cohereService: CohereService,
    private loadingCtrl: LoadingController,
    private toastCtrl: ToastController,
    private languageService: LanguageService,
    private translateService: TranslateService
    
  ) { 
    this.isMobile = this.platform.is('ios') || this.platform.is('android');
    this.formGroup = this.fb.group({
      name:['', [Validators.required, Validators.minLength(2)]],
      synopsis:['', [Validators.required]],
      releaseYear:['', [Validators.required]],
      rating:['', [Validators.required]],
      picture: ['']
      
    });
  }
  
  async generateSynopsis() {
    const title = this.formGroup.get('name')?.value;
    const languague = this.languageService.getStoredLanguage();
    
    if (!title) {
      const toast = await this.toastCtrl.create({
        message: await lastValueFrom(this.translateService.get('IA.NAME_REQUIRED')),
        duration: 2000,
        color: 'warning'
      });
      await toast.present();
      return;
    }

    const loading = await this.loadingCtrl.create({
      message: await lastValueFrom(this.translateService.get('IA.GENERATING')),
      spinner: 'circles'
    });
    await loading.present();

    this.isGeneratingSynopsis = true;
    
    this.cohereService.generateSynopsis(title, languague).subscribe({
      next: (synopsis) => {
        this.formGroup.get('synopsis')?.setValue(synopsis);
        this.formGroup.get('synopsis')?.markAsDirty();
        loading.dismiss();
        this.isGeneratingSynopsis = false;
      },
      error: async (error) => {
        console.error('Error al generar sinopsis:', error);
        const errorMessage = await lastValueFrom(this.translateService.get('IA.ERROR'));
        this.showErrorToast(errorMessage);
        loading.dismiss();
        this.isGeneratingSynopsis = false;
      }
    });
  }

  async showErrorToast(message: string) {
    const toast = await this.toastCtrl.create({
      message: message,
      duration: 3000,
      color: 'danger'
    });
    await toast.present();
  }

  ngOnInit() {} 

  get name(){
    return this.formGroup.controls['name'];
  }

  get synopsis(){
    return this.formGroup.controls['synopsis'];
  }

  get releaseYear(){
    return this.formGroup.controls['releaseYear'];
  }

  get rating(){
    return this.formGroup.controls['rating'];
  }
  
  getDirtyValues(formGroup: FormGroup): any {
    const dirtyValues: any = {};
  
    Object.keys(formGroup.controls).forEach(key => {
      const control = formGroup.get(key);
      if (control?.dirty) {
        if (key === 'picture' && control.value) {
          dirtyValues[key] = {
            url: control.value
          };
        } else {
          dirtyValues[key] = control.value;
        }
      }
    });
  
    return dirtyValues;
  }

  onSubmit(){
    if (this.formGroup.valid) {
      const formData = this.mode === 'new' ? 
        {
          ...this.formGroup.value,
          picture: this.formGroup.get('picture')?.value ? {
            url: this.formGroup.get('picture')?.value
          } : undefined
        } : 
        this.getDirtyValues(this.formGroup);
  
      this.modalCtrl.dismiss(formData, this.mode);
    }
  }

  dismiss() {
    this.modalCtrl.dismiss();
  }

}
