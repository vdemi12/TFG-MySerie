import { Component, OnInit, ViewChild } from '@angular/core';
import { ActivatedRoute } from '@angular/router';
import { Serie } from '../../core/models/serie.model';
import { SeriesService } from '../../core/services/impl/series.service';
import { IonicModule, IonTextarea, ToastController } from '@ionic/angular';
import { CommonModule } from '@angular/common';
import { TranslateModule, TranslateService } from '@ngx-translate/core';
import { ReactiveFormsModule, FormsModule } from '@angular/forms';
import { BehaviorSubject } from 'rxjs';
import { Comment } from '../../core/models/comment.model';
import { CommentsService } from '../../core/services/impl/comments.service';
import { BaseAuthenticationService } from '../../core/services/impl/base-authentication.service';
import { User } from '../../core/models/auth.model';
import { getFirestore, doc, getDoc } from 'firebase/firestore';

@Component({
  selector: 'app-comments-page',
  templateUrl: './comments.page.html',
  styleUrls: ['./comments.page.scss'],
  standalone: true,
  imports: [CommonModule, IonicModule, TranslateModule, ReactiveFormsModule, FormsModule]
})
export class CommentsPage implements OnInit {
  @ViewChild('commentTextarea') commentTextarea!: IonTextarea;
  
  serie?: Serie;
  isLoading: boolean = true;
  error: boolean = false;
  
  commentText: string = '';
  comments$ = new BehaviorSubject<Comment[]>([]);
  displayedComments: (Comment & {user?: User})[] = [];
  
  selectedReaction: 'like' | 'dislike' | null = null;
  
  likesCount: number = 0;
  dislikesCount: number = 0;
  
  currentPage: number = 1;
  pageSize: number = 10;
  hasMoreComments: boolean = false;
  loadingMoreComments: boolean = false;
  
  currentUser$ = new BehaviorSubject<User | undefined>(undefined);
  
  private usersMap: {[userId: string]: User} = {};

  constructor(
    private route: ActivatedRoute,
    private seriesService: SeriesService,
    private commentsService: CommentsService,
    private authService: BaseAuthenticationService,
    private toastCtrl: ToastController,
    private translate: TranslateService
  ) { }

  ngOnInit() {
    const serieId = this.route.snapshot.paramMap.get('id');
    if (serieId) {
      this.loadSerie(serieId);
      this.loadCurrentUser();
      
      this.comments$.subscribe(comments => {
        this.updateDisplayedComments(comments);
        this.updateReactionCounts(comments);
      });
    } else {
      this.error = true;
      this.isLoading = false;
    }
  }
  
  updateReactionCounts(comments: Comment[]) {
    this.likesCount = comments.filter(c => c.reaction === 'like').length;
    this.dislikesCount = comments.filter(c => c.reaction === 'dislike').length;
  }
  
  updateDisplayedComments(comments: Comment[]) {
    this.displayedComments = comments.map(comment => ({
      ...comment,
      user: this.usersMap[comment.userId]
    }));
  }
  
  async loadCurrentUser() {
    try {
      const user = await this.authService.getCurrentUser();
      this.currentUser$.next(user);
      if (user) {
        this.usersMap[user.id] = user;
      }
    } catch (error) {
      console.error('Error loading current user:', error);
    }
  }

  loadSerie(id: string) {
    this.isLoading = true;
    this.seriesService.getById(id).subscribe({
      next: (serie) => {
        if (serie) {
          this.serie = serie;
          this.error = false;
          this.loadComments(id);
        } else {
          this.error = true;
          this.isLoading = false;
        }
      },
      error: (err) => {
        console.error('Error loading serie:', err);
        this.error = true;
        this.isLoading = false;
      }
    });
  }
  
  loadComments(serieId: string) {
    this.isLoading = true;
    this.commentsService.getCommentsBySerieId(serieId, this.currentPage, this.pageSize).subscribe({
      next: (comments) => {
        const sortedComments = [...comments].sort((a, b) => 
          new Date(b.createdAt).getTime() - new Date(a.createdAt).getTime()
        );
        
        this.comments$.next(sortedComments);
        
        this.hasMoreComments = comments.length >= this.pageSize;
        
        this.loadUserData(sortedComments);
      },
      error: (err) => {
        console.error('Error al cargar comentarios:', err);
        this.isLoading = false;
      }
    });
  }
  
  loadMoreComments() {
    if (!this.serie || this.loadingMoreComments || !this.hasMoreComments) {
      return;
    }
    
    this.loadingMoreComments = true;
    this.currentPage++;
    
    this.commentsService.getCommentsBySerieId(this.serie.id, this.currentPage, this.pageSize).subscribe({
      next: (moreComments) => {
        const sortedMoreComments = [...moreComments].sort((a, b) => 
          new Date(b.createdAt).getTime() - new Date(a.createdAt).getTime()
        );
        
        const allComments = [...this.comments$.value, ...sortedMoreComments];
        this.comments$.next(allComments);
        
        this.hasMoreComments = moreComments.length >= this.pageSize;
        
        this.loadUserData(sortedMoreComments);
        this.loadingMoreComments = false;
      },
      error: (err) => {
        console.error('Error al cargar más comentarios:', err);
        this.loadingMoreComments = false;
      }
    });
  }
  
  loadUserData(comments: Comment[]) {
    if (comments.length === 0) {
      this.isLoading = false;
      return;
    }
    
    this.isLoading = true;
    
    const loadingPromises = comments.map(async (comment) => {
      if (comment.userId && !this.usersMap[comment.userId]) {
        try {
          const db = getFirestore();
          const userRef = doc(db, 'users', comment.userId);
          const userSnap = await getDoc(userRef);
          
          if (userSnap.exists()) {
            const userData = userSnap.data();
            this.usersMap[comment.userId] = {
              id: comment.userId,
              username: userData['username'] || 'Usuario',
              email: userData['email'] || '',
              name: userData['name'] || '',
              surname: userData['surname'] || '',
              admin: userData['admin'] || false,
              picture: userData['picture'] || undefined
            };
            
            if (!this.usersMap[comment.userId].name && this.usersMap[comment.userId].username) {
              this.usersMap[comment.userId].name = this.usersMap[comment.userId].username;
            }
          }
        } catch (error) {
          console.error('Error al cargar datos del usuario:', error);
        }
      }
    });
    
    Promise.all(loadingPromises).then(() => {
      this.updateDisplayedComments(comments);
      this.isLoading = false;
    }).catch(error => {
      console.error('Error al cargar datos de usuarios:', error);
      this.updateDisplayedComments(comments);
      this.isLoading = false;
    });
  }
  
  selectReaction(reaction: 'like' | 'dislike') {
    if (this.selectedReaction === reaction) {
      this.selectedReaction = null;
    } else {
      this.selectedReaction = reaction;
    }
  }
  
  async postComment() {
    if (!this.serie || !this.commentText || this.commentText.trim() === '' || !this.selectedReaction) {
      return;
    }
    
    const currentUser = this.currentUser$.value;
    if (!currentUser) {
      return;
    }
    
    const newComment: Comment = {
      id: '', 
      text: this.commentText.trim(),
      reaction: this.selectedReaction,
      serieId: this.serie.id,
      userId: currentUser.id,
      createdAt: new Date().toISOString()
    };
    
    this.commentsService.add(newComment).subscribe({
      next: async (addedComment) => {
        const updatedComments = [addedComment, ...this.comments$.value];
        this.comments$.next(updatedComments);
        
        this.commentText = '';
        this.selectedReaction = null;
        
        const toast = await this.toastCtrl.create({
          message: await this.translate.get('COMMENTS.ADDED_SUCCESSFULLY').toPromise(),
          duration: 3000,
          position: 'bottom',
          color: 'success'
        });
        await toast.present();
      },
      error: async (err) => {
        console.error('Error adding comment:', err);
        const toast = await this.toastCtrl.create({
          message: await this.translate.get('COMMENTS.ERROR_ADDING').toPromise(),
          duration: 3000,
          position: 'bottom',
          color: 'danger'
        });
        await toast.present();
      }
    });
  }
  
  trackByCommentId(index: number, comment: any): string {
    return comment.id;
  }
}
