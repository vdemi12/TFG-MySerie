import { Component, OnInit } from '@angular/core';
import { FormBuilder, FormGroup, Validators } from '@angular/forms';
import { ActivatedRoute, Router } from '@angular/router';
import { User } from 'src/app/core/models/auth.model';
import { ToastController } from '@ionic/angular';
import { BaseAuthenticationService } from 'src/app/core/services/impl/base-authentication.service';
import {
  passwordsMatchValidator,
  passwordValidator,
} from 'src/app/core/utils/validators';

@Component({
  selector: 'app-register',
  templateUrl: './register.page.html',
  styleUrls: ['./register.page.scss'],
})
export class RegisterPage implements OnInit {
  registerForm: FormGroup;
  passwordFieldType: string = 'password';
  passwordFieldType2: string = 'password';
  isAdmin: boolean = false;
  adminKeyFieldType: string = 'password';

  constructor(
    private fb: FormBuilder,
    private router: Router,
    private route: ActivatedRoute,
    private authSvc: BaseAuthenticationService,
    private toastController: ToastController
  ) {
    this.registerForm = this.fb.group(
      {
        username: ['', [Validators.required, Validators.minLength(2)]],
        name: ['', [Validators.required, Validators.minLength(2)]],
        surname: ['', [Validators.required, Validators.minLength(2)]],
        email: ['', [Validators.required, Validators.email]],
        password: ['', [Validators.required, passwordValidator]],
        confirmPassword: ['', [Validators.required]],
        adminKey: [''],
      },
      {
        validators: passwordsMatchValidator,
      }
    );
  }

  private adminKeyHash =
    'a595ad2f4500de5469d684bbbc62562456ff6990cf32231610b3deff162ce825'; //adminSecret123

  ngOnInit(): void {
    this.registerForm.get('adminKey')?.clearValidators();
  }

  setAdmin(value: boolean) {
    this.isAdmin = value;
    const adminKeyControl = this.registerForm.get('adminKey');
    if (value) {
      adminKeyControl?.setValidators([Validators.required]);
    } else {
      adminKeyControl?.clearValidators();
      adminKeyControl?.setValue('');
    }
    adminKeyControl?.updateValueAndValidity();
  }

  togglePasswordVisibility() {
    this.passwordFieldType =
      this.passwordFieldType === 'password' ? 'text' : 'password';
  }

  togglePasswordVisibility2() {
    this.passwordFieldType2 =
      this.passwordFieldType2 === 'password' ? 'text' : 'password';
  }

  toggleAdminKeyVisibility() {
    this.adminKeyFieldType =
      this.adminKeyFieldType === 'password' ? 'text' : 'password';
  }

  async presentToast(message: string, color: 'success' | 'danger') {
    const toast = await this.toastController.create({
      message: message,
      duration: 2000,
      position: 'bottom',
      color: color,
      buttons: [
        {
          icon: 'close-outline',
          role: 'cancel',
        },
      ],
    });
    await toast.present();
  }

  async onSubmit() {
    if (this.registerForm.valid) {
      const formValue = this.registerForm.value;
      const wantsAdmin = this.isAdmin;

      if (wantsAdmin) {
        const enteredKey = formValue.adminKey?.trim() || '';
        const enteredKeyHash = await this.hashString(enteredKey);
        console.log(enteredKeyHash)

        if (enteredKeyHash !== this.adminKeyHash) {
          this.presentToast('Clave de admin incorrecta', 'danger');
          return;
        }
      }


      const signUpData = {
        username: formValue.username,
        email: formValue.email,
        password: formValue.password,
        name: formValue.name,
        surname: formValue.surname,
        admin: wantsAdmin,
        adminKey: wantsAdmin ? formValue.adminKey : undefined,
      };

      this.authSvc.signUp(signUpData).subscribe({
        next: (resp: User) => {
          this.presentToast('¡Registro exitoso!', 'success');
          this.router.navigate(['/home']);
        },
        error: (err) => {
          console.error(err);
          this.presentToast('Error al registrarse', 'danger');
        },
      });
    }
  }

  onLogin() {
    this.registerForm.reset();
    const returnUrl = this.route.snapshot.queryParams['returnUrl'] || '/home';
    this.router.navigate(['/login'], {
      queryParams: { returnUrl: returnUrl },
      replaceUrl: true,
    });
  }

  async hashString(str: string): Promise<string> {
    const buffer = new TextEncoder().encode(str);
    const hashBuffer = await crypto.subtle.digest('SHA-256', buffer);
    const hashArray = Array.from(new Uint8Array(hashBuffer));
    return hashArray.map((b) => b.toString(16).padStart(2, '0')).join('');
  }

  get username() {
    return this.registerForm.controls['username'];
  }

  get name() {
    return this.registerForm.controls['name'];
  }

  get surname() {
    return this.registerForm.controls['surname'];
  }

  get email() {
    return this.registerForm.controls['email'];
  }

  get password() {
    return this.registerForm.controls['password'];
  }

  get confirmPassword() {
    return this.registerForm.controls['confirmPassword'];
  }

  get adminKey() {
    return this.registerForm.controls['adminKey'];
  }
}
