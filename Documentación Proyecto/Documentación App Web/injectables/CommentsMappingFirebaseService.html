import { Inject, Injectable } from "@angular/core";
import { IBaseMapping } from "../intefaces/base-mapping.interface";
import { Paginated } from "../../models/paginated.model";
import { Comment } from "../../models/comment.model";
import { doc, DocumentReference, Firestore, getFirestore } from "firebase/firestore";
import { FIREBASE_CONFIG_TOKEN } from "../repository.tokens";
import { initializeApp } from "firebase/app";

export interface FirebaseComment {
  text: string;
  reaction: 'like' | 'dislike' | null;
  serieId: DocumentReference;
  userId: string;
  createdAt: string;
}

@Injectable({
  providedIn: 'root'
})
export class CommentsMappingFirebaseService implements IBaseMapping<Comment> {
    
  private db: Firestore;
  constructor(@Inject(FIREBASE_CONFIG_TOKEN) protected firebaseConfig: any){
    this.db = getFirestore(initializeApp(firebaseConfig));
  }

  setAdd(data: Comment): FirebaseComment {
    let dataMapping: FirebaseComment = {
      text: data.text,
      reaction: data.reaction,
      serieId: doc(this.db, 'series', data.serieId),
      userId: data.userId || '',
      createdAt: data.createdAt || new Date().toISOString()
    };
    return dataMapping;
  }
  
  setUpdate(data: Partial<Comment>): Partial<FirebaseComment> {
    const result: any = {};
    
    if (data.text !== undefined) result.text = data.text;
    if (data.reaction !== undefined) result.reaction = data.reaction;
    if (data.serieId !== undefined) result.serieId = doc(this.db, 'series', data.serieId);
    if (data.userId !== undefined) result.userId = data.userId;
    
    return result;
  }
  
  getOne(data: { id: string } & FirebaseComment): Comment {
    return {
      id: data.id,
      text: data.text,
      reaction: data.reaction,
      serieId: data.serieId?.id,
      userId: data.userId,
      createdAt: data.createdAt
    };
  }
  
  getPaginated(page: number, pageSize: number, total: number, data: ({id:string} & FirebaseComment)[]): Paginated<Comment> {
    return {
      page,
      pageSize,
      pages: Math.ceil(total / pageSize),
      data: data.map(item => this.getOne(item))
    };
  }
  
  getAdded(data: {id:string} & FirebaseComment): Comment {
    return this.getOne(data);
  }
  
  getUpdated(data: {id:string} & FirebaseComment): Comment {
    return this.getOne(data);
  }
  
  getDeleted(data: {id:string} & FirebaseComment): Comment {
    return this.getOne(data);
  }
}
