import { Inject, Injectable } from '@angular/core';
import { initializeApp } from 'firebase/app';
import { 
  getFirestore, 
  collection, 
  doc, 
  getDoc, 
  getDocs, 
  addDoc, 
  updateDoc, 
  deleteDoc,
  query,
  where,
  limit,
  startAfter,
  orderBy,
  QueryConstraint
} from 'firebase/firestore';
import { from, map, Observable, mergeMap } from 'rxjs';
import { Comment } from '../../models/comment.model';
import { Paginated } from '../../models/paginated.model';
import { COMMENTS_REPOSITORY_MAPPING_TOKEN, FIREBASE_CONFIG_TOKEN } from '../repository.tokens';
import { IBaseMapping } from '../intefaces/base-mapping.interface';
import { ICommentsRepository } from '../intefaces/comments-repository.interface';
import { SearchParams } from '../intefaces/base-repository.interface';

@Injectable({
  providedIn: 'root'
})
export class CommentsRepositoryFirebaseService implements ICommentsRepository {
  private db;
  private collectionRef;
  
  constructor(
    @Inject(FIREBASE_CONFIG_TOKEN) protected firebaseConfig: any,
    @Inject(COMMENTS_REPOSITORY_MAPPING_TOKEN) protected mapping: IBaseMapping<Comment>
  ) {
    const app = initializeApp(firebaseConfig);
    this.db = getFirestore(app);
    this.collectionRef = collection(this.db, 'comments');
  }

  getCommentsBySerieId(serieId: string, page: number = 1, pageSize: number = 10): Observable<Comment[]> {
    const serieRef = doc(this.db, 'series', serieId);
    const q = query(
      this.collectionRef,
      where('serieId', '==', serieRef),
      orderBy('createdAt', 'desc'),
      limit(pageSize)
    );

    return from(getDocs(q)).pipe(
      map(snapshot => {
        const comments = snapshot.docs.map(doc => ({
          id: doc.id,
          ...doc.data()
        }));
        return comments.map(comment => this.mapping.getOne(comment as any));
      })
    );
  }

  private async getLastDocumentOfPreviousPage(page: number, pageSize: number, filters: SearchParams = {}) {
    if (page <= 1) return null;
    
    const constraints: QueryConstraint[] = [
      orderBy('createdAt', 'desc'),
      limit((page - 1) * pageSize)
    ];
    
    if (filters['serieId']) {
      const serieRef = doc(this.db, 'series', filters['serieId']);
      constraints.push(where('serieId', '==', serieRef));
    }
    
    const q = query(this.collectionRef, ...constraints);
    const snapshot = await getDocs(q);
    const docs = snapshot.docs;
    return docs[docs.length - 1];
  }

  getAll(page: number = 1, pageSize: number = 10, filters: SearchParams = {}): Observable<Comment[] | Paginated<Comment>> {
    return from(this.getLastDocumentOfPreviousPage(page, pageSize, filters)).pipe(
      map(lastDoc => {
        const constraints: QueryConstraint[] = [
          orderBy('createdAt', 'desc'),
          limit(pageSize)
        ];
        
        if (lastDoc) {
          constraints.push(startAfter(lastDoc));
        }
        
        if (filters['serieId']) {
          const serieRef = doc(this.db, 'series', filters['serieId']);
          constraints.push(where('serieId', '==', serieRef));
        }
        
        return query(this.collectionRef, ...constraints);
      }),
      mergeMap(q => getDocs(q)),
      map(snapshot => {
        const items = snapshot.docs.map(doc => ({
          id: doc.id,
          ...doc.data()
        }));
        return this.mapping.getPaginated(page, pageSize, items.length, items as any[]);
      })
    );
  }

  getById(id: string): Observable<Comment | null> {
    const docRef = doc(this.db, 'comments', id);
    return from(getDoc(docRef)).pipe(
      map(doc => {
        if (doc.exists()) {
          return this.mapping.getOne({ id: doc.id, ...doc.data() } as any);
        }
        return null;
      })
    );
  }

  add(entity: Comment): Observable<Comment> {
    return from(addDoc(this.collectionRef, this.mapping.setAdd(entity))).pipe(
      map(docRef => this.mapping.getAdded({ ...entity, id: docRef.id } as any))
    );
  }

  update(id: string, entity: Comment): Observable<Comment> {
    const docRef = doc(this.db, 'comments', id);
    
    return from(updateDoc(docRef, this.mapping.setUpdate(entity))).pipe(
      map(() => this.mapping.getUpdated({ ...entity, id } as any))
    );
  }

  delete(id: string): Observable<Comment> {
    const docRef = doc(this.db, 'comments', id);
    return from(getDoc(docRef)).pipe(
      map(doc => ({ id: doc.id, ...doc.data() } as any)),
      mergeMap(entity => {
        return from(deleteDoc(docRef)).pipe(
          map(() => this.mapping.getDeleted(entity))
        );
      })
    );
  }
}
